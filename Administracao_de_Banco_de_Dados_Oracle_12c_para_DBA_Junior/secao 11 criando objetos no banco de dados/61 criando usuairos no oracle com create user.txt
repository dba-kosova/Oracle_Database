61 criando usuairos no oracle com create user

obvervações ao criar um usuario no banco 
para colocar os objetos nele 



username 


authentication method 
->database(na maioria das vezes)
-> sistema operacional
-> network 
-> global 
-> serviços externos (LDAP)








basic privileges 




default permanent tablespace and space quotas 



default temporary tablespace 








CREATE USER ALUNO IDENTIFIED BY A1Un0#$
DEFAULT TABLESPACE USERS
TEMPORARY TABLESPACE TEMP 
QUOTA UNLIMITED ON USERS;






PARA QUE ELE CONSIGA SE CONECTAR E CRIAR ALGUNS OBJETOS 


GRANT CREATE SESSION TO ALUNO;

GRANT CREATE TABLE TO ALUNO;
