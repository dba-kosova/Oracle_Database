15- criando os usaurios para instalacao




groupadd -g 54321 oinstall
groupadd -g 54322 dba
groupadd -g 54323 oper 
groupadd -g 54327 asmdba 
groupadd -g 54328 asmoper
groupadd -g 54329 asmadmin




criar os usuarios 



useradd -u 54321 -g oinstall -G dba,oper,asmdba,asmadmin oracle 
useradd -u 54322 -g oinstall -G dba,oper,asmdba,asmadmin,asmoper grid 




criar o diretorio home 
onde fica a variavel de ambiente 
/home/oracle 
/home/grid 




mkdir -p /home/oracle 
mkdir -p /home/grid 


chown oracle.dba /home/oracle
chown grid.asmdba /home/grid


chmod 775 /home/oracle

chmod 775 /home/grid
